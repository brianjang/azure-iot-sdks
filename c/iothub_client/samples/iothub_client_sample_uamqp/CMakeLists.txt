#this is CMakeLists.txt for iothub_client_sample_amqp

compileAsC99()

if(NOT ${use_uamqp})
	message(FATAL_ERROR "iothub_client_sample_uamqp being generated without uamqp support")
endif()

set(iothub_client_sample_uamqp_c_files
	iothub_client_sample_uamqp.c
)

if(WIN32)
	set(iothub_client_sample_uamqp_c_files ${iothub_client_sample_uamqp_c_files} ./windows/main.c)
else()
	set(iothub_client_sample_uamqp_c_files ${iothub_client_sample_uamqp_c_files} ./linux/main.c)
endif()

set(iothub_client_sample_uamqp_h_files
	iothub_client_sample_uamqp.h
)

IF(WIN32)
	#windows needs this define
	add_definitions(-D_CRT_SECURE_NO_WARNINGS)
	add_definitions(-DGB_MEASURE_MEMORY_FOR_THIS -DGB_DEBUG_ALLOC)
ENDIF(WIN32)

include_directories(.)

add_executable(iothub_client_sample_uamqp ${iothub_client_sample_uamqp_c_files} ${iothub_client_sample_uamqp_h_files})

target_link_libraries(iothub_client_sample_uamqp
		iothub_client
		iothub_client_uamqp_transport)

linkSharedUtil(iothub_client_sample_uamqp)
linkUAMQP(iothub_client_sample_uamqp)
